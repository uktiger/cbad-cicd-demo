name: CI/CD Pipeline

on:
  push:
    branches:
      - main

jobs:
  build-test-push:
    runs-on: ubuntu-latest

    steps:

    - name: Checkout code
      uses: actions/checkout@v3

    - name: Security Scan (Hardcoded Secrets)
      run: |
        echo "Scanning for hardcoded secrets..."
        if grep -i "password" Dockerfile; then
          echo "❌ Hardcoded password found in Dockerfile"
          exit 1
        else
          echo "✅ No hardcoded password found"
        fi

    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v2

    - name: Login to Docker Hub
      uses: docker/login-action@v2
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_PASSWORD }}

    - name: Build Docker Image
      run: docker build -t ${{ secrets.DOCKER_USERNAME }}/hello-cicd:latest .

    - name: Run Container (Test)
      run: |
        docker run -d -p 8080:80 ${{ secrets.DOCKER_USERNAME }}/hello-cicd:latest
        sleep 5
        docker ps

    - name: Push Docker Image
      run: docker push ${{ secrets.DOCKER_USERNAME }}/hello-cicd:latest
